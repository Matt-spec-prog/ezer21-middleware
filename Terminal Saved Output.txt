        3 // Pushes all transformed financial data to Base44 using the REST API directly.
        4 // (The @base44/sdk package is ESM-only; our project is CommonJS, so we call
        5 // the same REST endpoints the SDK wraps.)
        6 //
        7 // Authentication: Bearer token from BASE44_API_KEY env var + X-App-Id header.
        8 // Base URL: https://base44.app/api
        9 //
       10 // Push strategy: "delete all for company + replace" per entity type.
       11 //   - Deletes all existing records for the company (by period_type where relevant)
       12 //   - Bulk-creates the new records
       13 //   This is safe for a scheduled middleware run and avoids duplicates.
       14 //
       15 // ForecastAssumptions: upserted once.
       16 //   - Created with system defaults on first push.
       17 //   - On subsequent pushes: system_defaults_json is updated (powers the
       18 //     "revert to defaults" button), but client-edited fields are not touched.

       70 // Delete all records matching a query, then bulk-create replacements in chunks.
       71 // Returns count of records created.
       72 async function replaceAll(http, entityName, deleteQuery, records) {
       73   // Delete all matching existing records
       74   await apiDelete(http, entityPath(entityName), deleteQuery);
       75
       76   if (records.length === 0) return 0;
       77
       78   // Bulk-create in chunks
       79   for (let i = 0; i < records.length; i += CHUNK_SIZE) {
       80     const chunk = records.slice(i, i + CHUNK_SIZE);
       81     await apiPost(http, `${entityPath(entityName)}/bulk`, chunk);
       82   }
       83
       84   return records.length;
       85 }
